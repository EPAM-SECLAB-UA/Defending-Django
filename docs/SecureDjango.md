### 2. Rate Limiting Middleware

```python
import redis
from django.http import JsonResponse
from django.conf import settings

class RateLimitMiddleware:
    def __init__(self, get_response):
        self.get_response = get_response
        self.redis_client = redis.Redis(
            host=settings.REDIS_HOST,
            port=settings.REDIS_PORT,
            db=settings.REDIS_DB,
            decode_responses=True
        )
        self.max_requests = getattr(settings, 'RATE_LIMIT_REQUESTS', 100)
        self.window_size = getattr(settings, 'RATE_LIMIT_WINDOW', 60)
    
    def __call__(self, request):
        # –û—Ç—Ä–∏–º–∞—Ç–∏ IP –∫–ª—ñ—î–Ω—Ç–∞
        client_ip = self.get_client_ip(request)
        
        # –°—Ç–≤–æ—Ä–∏—Ç–∏ —É–Ω—ñ–∫–∞–ª—å–Ω–∏–π –∫–ª—é—á
        rate_limit_key = f"rate_limit:{client_ip}"
        
        try:
            # –û—Ç—Ä–∏–º–∞—Ç–∏ –ø–æ—Ç–æ—á–Ω—É –∫—ñ–ª—å–∫—ñ—Å—Ç—å –∑–∞–ø–∏—Ç—ñ–≤
            current_requests = self.redis_client.get(rate_limit_key)
            
            if current_requests is None:
                # –ü–µ—Ä—à–∏–π –∑–∞–ø–∏—Ç —É —Ü—å–æ–º—É –≤—ñ–∫–Ω—ñ
                pipe = self.redis_client.pipeline()
                pipe.set(rate_limit_key, 1, ex=self.window_size)
                pipe.execute()
                current_requests = 1
            else:
                current_requests = int(current_requests)
                
                if current_requests >= self.max_requests:
                    # –õ—ñ–º—ñ—Ç –ø–µ—Ä–µ–≤–∏—â–µ–Ω–æ
                    return JsonResponse({
                        'error': 'Rate limit exceeded',
                        'detail': f'Maximum {self.max_requests} requests per {self.window_size} seconds'
                    }, status=429)
                
                # –ó–±—ñ–ª—å—à–∏—Ç–∏ –ª—ñ—á–∏–ª—å–Ω–∏–∫
                self.redis_client.incr(rate_limit_key)
            
            # –î–æ–¥–∞—Ç–∏ –∑–∞–≥–æ–ª–æ–≤–∫–∏ rate limit –¥–æ –≤—ñ–¥–ø–æ–≤—ñ–¥—ñ
            response = self.get_response(request)
            response['X-RateLimit-Limit'] = str(self.max_requests)
            response['X-RateLimit-Remaining'] = str(self.max_requests - current_requests)
            
            return response
            
        except redis.RedisError:
            # –Ø–∫—â–æ Redis –Ω–µ –ø—Ä–∞—Ü—é—î, –ø—Ä–æ–ø—É—Å–∫–∞—î–º–æ –∑–∞–ø–∏—Ç (fail open)
            return self.get_response(request)
    
    def get_client_ip(self, request):
        """–í–∏—Ç—è–≥–Ω—É—Ç–∏ IP-–∞–¥—Ä–µ—Å—É –∫–ª—ñ—î–Ω—Ç–∞"""
        x_forwarded_for = request.META.get('HTTP_X_FORWARDED_FOR')
        if x_forwarded_for:
            return x_forwarded_for.split(',')[0].strip()
        return request.META.get('REMOTE_ADDR')
```
